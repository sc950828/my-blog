    # 1、队列

    队列是先入先出的数据结构。添加始终在队尾，删除始终在队头。有普通队列、优先队列、循环队列

    class Queue {
  constructor() {
    this.items = [];
  }
  // 入队
  enqueue(item) {
    this.items.push(item);
  }

  // 优先队列入队
  priorityenqueue(item, priority) {
    const priorityItem = { item, priority };
    if (this.isEmpty()) {
      this.items.push(priorityItem);
    } else {
      // 是否优先级有大于里面元素
      const preIndex = this.items.findIndex(
        item => priorityItem.priority > item.priority
      );
      if (preIndex > -1) {
        // 有大于插入到索引位置
        this.items.splice(preIndex, 0, priority);
      } else {
        // 没有添加到最后
        this.items.push(priorityItem);
      }
    }
  }
  // 出队
  dequeue() {
    this.items.shift();
  }
  // 寻找
  find(index) {
    return !this.isEmpty()
      ? index >= this.size()
        ? null
        : this.items[index]
      : null;
  }
  // 循环队列查找
  loopfind(index) {
    return !this.isEmpty()
      ? index > this.size()
        ? this.items[index % this.size()]
        : this.items[index]
      : null;
  }
  // 清空
  clear() {
    this.items = [];
  }
  // 大小长度
  size() {
    return this.items.length;
  }
  // 判空
  isEmpty() {
    return !this.items.length;
  }
  // 打印
  print() {
    console.log(this.items.toString());
  }
}

    # 2、栈

    栈是先入后出的数据结构。新增删除始终在栈尾。

    class Stack {
  constructor() {
    this.items = [];
  }
  // 入栈
  push(item) {
    this.items.push(item);
  }
  // 出栈
  pop() {
    return this.items.pop();
  }
  // 是否为空
  isEmpty() {
    return !this.items.length;
  }
  // 长度大小
  size() {
    return this.items.length;
  }
  // 清空
  clear() {
    this.items = [];
  }
  // 打印
  print() {
    console.log(this.items.toString());
  }
}

    # 3、使用两个栈实现一个队列

    // 队列是先入后出
// 一个栈放数据
// 一个栈拿数据 没数据的时候把栈1的数据全拿过来
const stack1 = [];
const stack2 = [];

function push(item) {
  stack1.push(item);
}

function pop() {
  if (stack2.length === 0) {
    while (stack1.length > 0) {
      stack2.push(stack1.pop());
    }
  }

  return stack2.pop() || null;
}

    # 使用两个队列实现一个栈

    // 栈都是从栈尾新增删除数据 先入后出
// 两个对列交替存数据 把最新的数据放队头
// 出栈 就是队列头先出 实现后入先出
const queue1 = [];
const queue2 = [];

function push(item) {
  if (queue1.length === 0) {
    // 把最新的数据放队头
    queue1.push(item);
    while (queue2.length > 0) {
      queue1.push(queue2.shift());
    }
  } else if (queue2.length === 0) {
    queue2.push(item);
    while (queue1.length > 0) {
      queue2.push(queue1.shift());
    }
  }
}
// 模拟出栈
function pop() {
  if (queue1.length > 0) {
    return queue1.shift();
  } else if (queue2.length > 0) {
    return queue2.shift();
  }
}
