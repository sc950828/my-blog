    # 正则大全库 持续更新

    any-rule

    # 视频(video)链接地址（视频格式可按需增删）

    /^https?:\/\/(.+\/)+.+(\.(swf|avi|flv|mpg|rm|mov|wav|asf|3gp|mkv|rmvb|mp4))$/i;

    # 图片(image)链接地址（图片格式可按需增删）

    /^https?:\/\/(.+\/)+.+(\.(gif|png|jpg|jpeg|webp|svg|psd|bmp|tif))$/i;

    # 中文姓名

    /^(?:[\u4e00-\u9fa5·]{2,16})$/;

    # 英文姓名

    /(^[a-zA-Z]{1}[a-zA-Z\s]{0,20}[a-zA-Z]{1}$)/;

    # 帐号是否合法(字母开头，允许 5-16 字节，允许字母数字下划线组合

    /^[a-zA-Z]\w{4,15}$/;

    # 用户名校验，4 到 16 位（字母，数字，下划线，中划线）

    /^[a-zA-Z0-9_-]{4,16}$/;

    # 微信号(wx)，6 至 20 位，以字母开头，字母，数字，减号，下划线

    /^[a-zA-Z][-_a-zA-Z0-9]{5,19}$/;

    # 密码强度校验，最少 6 位，包括至少 1 个大写字母，1 个小写字母，1 个数字，1 个特殊字符

    /^\S*(?=\S{6,})(?=\S*\d)(?=\S*[A-Z])(?=\S*[a-z])(?=\S*[!@#$%^&*? ])\S*$/;

    # 手机号(mobile phone)中国(严谨), 根据工信部 2019 年最新公布的手机号段

    /^(?:(?:\+|00)86)?1(?:(?:3[\d])|(?:4[5-7|9])|(?:5[0-3|5-9])|(?:6[5-7])|(?:7[0-8])|(?:8[\d])|(?:9[1|8|9]))\d{8}$/;

    # 手机号(mobile phone)中国(宽松), 只要是 13,14,15,16,17,18,19 开头即可

    /^(?:(?:\+|00)86)?1[3-9]\d{9}$/;

    # 手机号(mobile phone)中国(最宽松), 只要是 1 开头即可, 如果你的手机号是用来接收短信, 优先建议选择这一条

    /^(?:(?:\+|00)86)?1\d{10}$/;

    # 座机(tel phone)电话(国内),如: 0341-86091234

    /^\d{3}-\d{8}$|^\d{4}-\d{7,8}$/;

    # email(邮箱)

    /^(([^<>()\[\]\\.,;:\s@"]+(\.[^<>()\[\]\\.,;:\s@"]+)*)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;

    # 身份证号(1 代,15 位数字)

    /^[1-9]\d{7}(?:0\d|10|11|12)(?:0[1-9]|[1-2][\d]|30|31)\d{3}$/;

    # 身份证号(2 代,18 位数字),最后一位是校验位,可能为数字或字符 X

    /^[1-9]\d{5}(?:18|19|20)\d{2}(?:0[1-9]|10|11|12)(?:0[1-9]|[1-2]\d|30|31)\d{3}[\dXx]$/;

    # 身份证号, 支持 1/2 代(15 位/18 位数字)

    /(^\d{8}(0\d|10|11|12)([0-2]\d|30|31)\d{3}$)|(^\d{6}(18|19|20)\d{2}(0[1-9]|10|11|12)([0-2]\d|30|31)\d{3}(\d|X|x)$)/;

    # qq 号格式正确

    /^[1-9][0-9]{4,10}$/;

    # 护照（包含香港、澳门）

    /(^[EeKkGgDdSsPpHh]\d{8}$)|(^(([Ee][a-fA-F])|([DdSsPp][Ee])|([Kk][Jj])|([Mm][Aa])|(1[45]))\d{7}$)/;

    # 中文/汉字

    /^(?:[\u3400-\u4DB5\u4E00-\u9FEA\uFA0E\uFA0F\uFA11\uFA13\uFA14\uFA1F\uFA21\uFA23\uFA24\uFA27-\uFA29]|[\uD840-\uD868\uD86A-\uD86C\uD86F-\uD872\uD874-\uD879][\uDC00-\uDFFF]|\uD869[\uDC00-\uDED6\uDF00-\uDFFF]|\uD86D[\uDC00-\uDF34\uDF40-\uDFFF]|\uD86E[\uDC00-\uDC1D\uDC20-\uDFFF]|\uD873[\uDC00-\uDEA1\uDEB0-\uDFFF]|\uD87A[\uDC00-\uDFE0])+$/;

    # 小数

    /^\d+\.\d+$/;

    # 数字

    /^\d{1,}$/;

    # 数字和字母组成

    /^[A-Za-z0-9]+$/;

    # 中文和数字

    /^((?:[\u3400-\u4DB5\u4E00-\u9FEA\uFA0E\uFA0F\uFA11\uFA13\uFA14\uFA1F\uFA21\uFA23\uFA24\uFA27-\uFA29]|[\uD840-\uD868\uD86A-\uD86C\uD86F-\uD872\uD874-\uD879][\uDC00-\uDFFF]|\uD869[\uDC00-\uDED6\uDF00-\uDFFF]|\uD86D[\uDC00-\uDF34\uDF40-\uDFFF]|\uD86E[\uDC00-\uDC1D\uDC20-\uDFFF]|\uD873[\uDC00-\uDEA1\uDEB0-\uDFFF]|\uD87A[\uDC00-\uDFE0])|(\d))+$/;

    # 英文字母

    /^[a-zA-Z]+$/;

    # 小写英文字母组成

    /^[a-z]+$/;

    # 大写英文字母

    /^[A-Z]+$/;

    # html 标签(宽松匹配)

    /<(\w+)[^>]*>(.*?<\/\1>)?/;

    # 车牌号(新能源)

    /[京津沪渝冀豫云辽黑湘皖鲁新苏浙赣鄂桂甘晋蒙陕吉闽贵粤青藏川宁琼使领 A-Z]{1}[A-HJ-NP-Z]{1}(([0-9]{5}[DF])|([DF][A-HJ-NP-Z0-9][0-9]{4}))$/;

    # 火车车次

    /^[GCDZTSPKXLY1-9]\d{1,4}$/;

    # 手机机身码(IMEI)

    /^\d{15,17}$/;

    # 必须带端口号的网址(或 ip)

    /^((ht|f)tps?:\/\/)?[\w-]+(\.[\w-]+)+:\d{1,5}\/?$/;

    # 网址(url,支持端口和"?+参数"和"#+参数)

    /^(((ht|f)tps?):\/\/)?[\w-]+(\.[\w-]+)+([\w.,@?^=%&:/~+#-]*[\w@?^=%&/~+#-])?$/;

    # 统一社会信用代码

    /^[0-9A-HJ-NPQRTUWXY]{2}\d{6}[0-9A-HJ-NPQRTUWXY]{10}$/;

    # 迅雷链接

    /^thunderx?:\/\/[a-zA-Z\d]+=$/;

    # ed2k 链接(宽松匹配)

    /^ed2k:\/\/\|file\|.+\|\/$/;

    # 磁力链接(宽松匹配)

    /^magnet:\?xt=urn:btih:[0-9a-fA-F]{40,}.*$/;

    # mac 地址

    /^((([a-f0-9]{2}:){5})|(([a-f0-9]{2}-){5}))[a-f0-9]{2}$/i;

    # 子网掩码

    /^(?:\d{1,2}|1\d\d|2[0-4]\d|25[0-5])(?:\.(?:\d{1,2}|1\d\d|2[0-4]\d|25[0-5])){3}$/;

    # ip-v4

    /^(?:(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$/;

    # ip-v6

    /^((([0-9A-Fa-f]{1,4}:){7}[0-9A-Fa-f]{1,4})|(([0-9A-Fa-f]{1,4}:){6}:[0-9A-Fa-f]{1,4})|(([0-9A-Fa-f]{1,4}:){5}:([0-9A-Fa-f]{1,4}:)?[0-9A-Fa-f]{1,4})|(([0-9A-Fa-f]{1,4}:){4}:([0-9A-Fa-f]{1,4}:){0,2}[0-9A-Fa-f]{1,4})|(([0-9A-Fa-f]{1,4}:){3}:([0-9A-Fa-f]{1,4}:){0,3}[0-9A-Fa-f]{1,4})|(([0-9A-Fa-f]{1,4}:){2}:([0-9A-Fa-f]{1,4}:){0,4}[0-9A-Fa-f]{1,4})|(([0-9A-Fa-f]{1,4}:){6}((\b((25[0-5])|(1\d{2})|(2[0-4]\d)|(\d{1,2}))\b)\.){3}(\b((25[0-5])|(1\d{2})|(2[0-4]\d)|(\d{1,2}))\b))|(([0-9A-Fa-f]{1,4}:){0,5}:((\b((25[0-5])|(1\d{2})|(2[0-4]\d)|(\d{1,2}))\b)\.){3}(\b((25[0-5])|(1\d{2})|(2[0-4]\d)|(\d{1,2}))\b))|(::([0-9A-Fa-f]{1,4}:){0,5}((\b((25[0-5])|(1\d{2})|(2[0-4]\d)|(\d{1,2}))\b)\.){3}(\b((25[0-5])|(1\d{2})|(2[0-4]\d)|(\d{1,2}))\b))|([0-9A-Fa-f]{1,4}::([0-9A-Fa-f]{1,4}:){0,5}[0-9A-Fa-f]{1,4})|(::([0-9A-Fa-f]{1,4}:){0,6}[0-9A-Fa-f]{1,4})|(([0-9A-Fa-f]{1,4}:){1,7}:))$/i;

    # linux 隐藏文件路径

    /^\/(?:[^/]+\/)*\.[^/]*/;

    /^\/(?:[^/]+\/)*[^/]+$/;

    # window 文件夹路径

    /^[a-zA-Z]:\\(?:\w+\\?)*$/;

    /^[a-zA-Z]:\\(?:\w+\\)*\w+\.\w+$/;

    # 股票代码(A 股)

    /^(s[hz]|S[HZ])(000[\d]{3}|002[\d]{3}|300[\d]{3}|600[\d]{3}|60[\d]{4})$/;

    # 大于等于 0, 小于等于 150, 支持小数位出现 5, 如 145.5, 用于判断考卷分数

    /^150$|^(?:\d|[1-9]\d|1[0-4]\d)(?:.5)?$/;

    # html 注释

    /^<!--[\s\S]*?-->$/;

    # md5 格式加密(32 位)

    /^([a-f\d]{32}|[A-F\d]{32})$/;

    # base64 格式

    /^\s*data:(?:[a-z]+\/[a-z0-9-+.]+(?:;[a-z-]+=[a-z0-9-]+)?)?(?:;base64)?,([a-z0-9!$&',()*+;=\-._~:@/?%\s]*?)\s*$/i;

    # 版本号(version)格式必须为 X.Y.Z

    /^\d+(?:\.\d+){2}$/;

    # 24 小时制时间（HH:mm:ss）

    /^(?:[01]\d|2[0-3]):[0-5]\d:[0-5]\d$/;

    # 12 小时制时间（hh:mm:ss）

    /^(?:1[0-2]|0?[1-9]):[0-5]\d:[0-5]\d$/;

    # date(日期)

    /^\d{4}(-)(1[0-2]|0?\d)\1([0-2]\d|\d|30|31)$/;

    # 数字/货币金额（支持负数、千分位分隔符）

    /^-?\d+(,\d{3})*(\.\d{1,2})?$/;

    # 数字/货币金额 (只支持正数、不支持校验千分位分隔符)

    /(?:^[1-9]([0-9]+)?(?:\.[0-9]{1,2})?$)|(?:^(?:0){1}$)|(?:^[0-9]\.[0-9](?:[0-9])?$)/;

    /(.)\1+/;
