### 1、call apply bind

```js
// call
function myCall(context) {
  if (typeof this !== "function") {
    throw new TypeError("type error");
  }
  let args = arguments.slice(1);
  let result = null;

  context = context || window;
  context.fn = this;

  result = context.fn(...args);

  delete context.fn;

  return result;
}
```

### 2、instanceof

```js
function myInstanceof(l, r) {
  let L = l.__proto__;
  let R = r.prototype;
  while (true) {
    if (L === R) {
      return true;
    }
    if (L === null || L === "undefined") {
      return false;
    }
    L = L.__proto__;
  }
}
```
